import { useRef } from 'react';

          className="text-center mb-40"

                        y: [0, -20, 0],
                        opacity: [0.3, 0.8, 0.3],
                        scale: [1, 1.5, 1]

                      transition={{
                        duration: 3,
                        repeat: Infinity,

                          animate={isInView ? { opacity: 1, y: 0 } : {}}
                          transition={{ 
                            duration: 0.8, 
                            delay: i * 0.2 + 0.7 + wordIndex * 0.1

                        </motion.span>

                        duration: 1, 
                        delay: i * 0.2 + 0.5,
                        type: "spring",

                      whileHover={{ 
                        scale: 1.2,

          <motion.div
            className="inline-block"
            whileHover={{ scale: 1.05 }}
